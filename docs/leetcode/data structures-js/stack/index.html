stack

定义​

JavaScript中的栈数据结构就是基于Array类型来进行封装的。 栈：类比手枪子弹。堆栈是正执行的函数。 应用场景：浏览器后退

区分​

访问：通过索引找到某元素的值， 搜索：直接找某元素，需要遍历，和索引无关，通过值找索引

基本方法的时间复杂度​

1.访问O(1) 栈顶元素 2.搜索O(N)

3.插入O(1)

4.删除O(1) 栈顶元素

特点​

调用栈，后进先出，先进后出，

练习题​

20有效的括号

//匹配左括号的时候，右括号先入栈，就只需要比较当前元素和栈顶相不相等就可以了，比左括号先入栈代码实现要简单的多了！
 var isValid = function (s) {
  const stack = [];
  if (stack.length % 2 != 0) return false; // 如果s的长度为奇数，一定不符合要求
  for (let i = 0; i < s.length; i++) {
    let c = s[i];
    switch (c) {
      case '(':
        stack.push(')');
        break;
      case '[':
        stack.push(']');
        break;
      case '{':
        stack.push('}');
        break;
      default://不是以上情况，如果是空或者是右括号判断是否和栈顶元素匹配
        if (stack.length === 0||c !== stack.pop()) {
          return false;
        }
    }
  }
  return stack.length === 0;
};
// 简化版本
var isValid = function(s) {
    const stack = [], 
        map = {
            "(":")",
            "{":"}",
            "[":"]"
        };
    for(const x of s) {
        if(x in map) {
            stack.push(x);
            continue;
        };
        if(map[stack.pop()] !== x) return false;
    }
    return !stack.length;
};
